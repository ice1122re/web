        async function fetchWithRetry(url, options, maxRetries = 3) {
            for (let i = 0; i < maxRetries; i++) {
                try {
                    const response = await fetch(url, options);

        let scene, camera, renderer, knot;
        let mouseX = 0, mouseY = 0;
        let windowHalfX = window.innerWidth / 2;
        let windowHalfY = window.innerHeight / 2;

        function initThree() {
            try {
                scene = new THREE.Scene();
                
                // Camera
                camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
                camera.position.z = 3;

                // Renderer
                const canvas = document.getElementById('three-bg-canvas');
                if (!canvas || typeof canvas.getContext !== 'function') {
                    console.error("Canvas element not found or is not a canvas.");
                    return; // Stop if canvas is invalid
                }
                renderer = new THREE.WebGLRenderer({ canvas: canvas, alpha: true, antialias: true });
                renderer.setSize(window.innerWidth, window.innerHeight);
                renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2)); // Optimize for performance

                // Object (Torus Knot as Points)
                const geometry = new THREE.TorusKnotGeometry(1, 0.3, 200, 32);
                const material = new THREE.PointsMaterial({
                    size: 0.01,
                    color: 0x8B5CF6, // --primary-color
                    transparent: true,
                    opacity: 0.8,
                    blending: THREE.AdditiveBlending
                });
                knot = new THREE.Points(geometry, material);
                scene.add(knot);

                // Lights (Subtle)
                const light1 = new THREE.PointLight(0xEC4899, 1, 100); // --accent-color
                light1.position.set(5, 5, 5);
                scene.add(light1);
                const light2 = new THREE.PointLight(0x6366F1, 1, 100); // --secondary-color
                light2.position.set(-5, -5, -5);
                scene.add(light2);

                // Event Listeners
                document.addEventListener('mousemove', onDocumentMouseMove);
                window.addEventListener('resize', onWindowResize);
                document.addEventListener('scroll', onDocumentScroll); // 10x Feature
                
                // Start animation
                animate();

            } catch (e) {
                console.error("Error in initThree:", e);
                // Hide 3D canvas if it fails
                const canvas = document.getElementById('three-bg-canvas');
                if(canvas) canvas.style.display = 'none';
            }
        }

        function onWindowResize() {
            windowHalfX = window.innerWidth / 2;
            windowHalfY = window.innerHeight / 2;
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
        }

        function onDocumentMouseMove(event) {
            mouseX = (event.clientX - windowHalfX);
            mouseY = (event.clientY - windowHalfY);
        }

        // 10x Feature: Link Scroll to 3D Rotation
        function onDocumentScroll() {
            const scrollY = window.scrollY;
            if (knot) {
                knot.rotation.y = scrollY * 0.001; // Twist on Y-axis
                knot.rotation.x = scrollY * 0.0005; // Tilt on X-axis
            }
        }

        // Animation Loop
        function animate() {
            if (!renderer) return; // Stop animation if renderer failed to init
            requestAnimationFrame(animate);

            // Mouse Interaction
            camera.position.x += (mouseX * 0.00005 - camera.position.x) * 0.05;
            camera.position.y += (-mouseY * 0.00005 - camera.position.y) * 0.05;
            camera.lookAt(scene.position);

            // Object Animation
            if (knot) {
                knot.rotation.z += 0.001; // Gentle constant spin
            }

            renderer.render(scene, camera);
        }

        // --- DOMContentLoaded Wrapper ---
        document.addEventListener('DOMContentLoaded', () => {
            // Start Typing
            typeWriter();
            
            // Add error handling for images
            document.querySelectorAll('.project-card img').forEach(img => {
                if (img.complete) return; // Skip if already loaded (e.g., from cache)
                const fallbackSrc = `https://placehold.co/600x400/1F2937/8B5CF6?text=${encodeURIComponent(img.alt || 'Image')}`;
                img.onerror = () => {
                    img.src = fallbackSrc;
                    img.onclick = null; // Disable modal on broken image
                };
            });
        });

        // --- Window Load Wrapper (for heavy assets like 3D) ---
        window.onload = () => {
            // Init 3D Scene
            initThree();
            // animate(); // Animate is called inside initThree now

            // Hide Loader
            const loader = document.getElementById('loader');
            loader.style.opacity = '0';
            setTimeout(() => {
                loader.style.display = 'none';
            }, 500); // Wait for fade-out
        };

    </script>
